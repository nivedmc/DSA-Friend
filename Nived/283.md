import java.util.*;
class Solution
{
    public void moveZeroes(int[] nums)
    {
        Scanner sc=new Scanner(System.in);
        int c=0;
        int arr[]=new int[nums.length];
            for(int i=0;i<nums.length;i++)
            {
                if(nums[i]!=0)
                {
                arr[c]=nums[i];
                    c++;
            }
               
            }
       for(int i=0;i<nums.length;i++)
       {
        nums[i]=arr[i];
        }
        
    
    }
}
